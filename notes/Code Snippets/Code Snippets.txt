//Wall slide
if (is_on_wall == true) {
	if (y_speed >  0) {
	x_speed = _gravity / 2;
	y_speed = min(y_speed + 1, 5);
		if (input_jump) { 
			x_speed += sign(facing) * 8;
			y_speed += _jump_height;
			}
	}
}

//Jump
if (input_jump) && (jump_current > 0){
	if(is_on_ground()) {
		y_speed += _jump_height;
		jump_current -= 1;
	} else {
		y_speed = dbl_jump_height;
		jump_current -= 1;
	}
}
